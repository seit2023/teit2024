void asc(int *a,int sz)
{
int i, j;
int temp;
for(i=0; i<sz; i++)
{
for(j=i+1; j<sz; j++)
{
if(a[i]>a[j])
{
temp = a[i];
a[i] = a[j];
a[j] = temp;
}
}
}
printf("\n Parent Process");
printf("\nSorted array in ascending order is:\n");
for(i=0; i<sz; i++)
{
printf("%d\t", a[i]) ;
}
